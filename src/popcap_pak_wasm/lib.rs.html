11struct PakState {
12    pak: popcap_pak::Pak<'static>,
13
14    /// Speed up lookups for name into idx
15    entry_path_idx_mapper: HashMap<Vec<u8>, Option<usize>>,
16}
17
18impl PakState {
19    /// Get the entry index by path
20    fn get_entry_idx_by_path(&mut self, path: Vec<u8>) -> Option<usize> {
21        match self.entry_path_idx_mapper.entry(path.clone()) {
22            HashMapEntry::Occupied(entry) => *entry.get(),
23            HashMapEntry::Vacant(entry) => {
24                let maybe_idx = self
25                    .pak
26                    .entries
27                    .iter()
28                    .position(|entry| entry.path() == path);
29                entry.insert(maybe_idx);
30                maybe_idx
31            }
32        }
33    }
34}

36#[wasm_bindgen]
37pub struct Pak {
38    state: Rc<RefCell<PakState>>,
39
40    /// Maps paths into entries.
41    entry_path_mapper: HashMap<Vec<u8>, Weak<EntryState>>,
42}
43
44#[wasm_bindgen]
45impl Pak {
46    /// Make a new Pak.
47    #[wasm_bindgen(constructor)]
48    pub fn new(file: &JsValue) -> Result<Pak, JsError> {
49        if file.is_undefined() {
50            return Ok(Self {
51                state: Rc::new(RefCell::new(PakState {
52                    pak: popcap_pak::Pak::new(),
53
54                    entry_path_idx_mapper: HashMap::new(),
55                })),
56
57                entry_path_mapper: HashMap::new(),
58            });
59        }
60
61        if let Some(arraybuffer) = file.dyn_ref::<ArrayBuffer>() {
62            let array = Uint8Array::new(arraybuffer).to_vec();
63            let pak = popcap_pak::Pak::from_read(array.as_slice())
64                .map_err(|e| JsError::new(&e.to_string()))?;
65            return Ok(Self {
66                state: Rc::new(RefCell::new(PakState {
67                    pak,
68
69                    entry_path_idx_mapper: HashMap::new(),
70                })),
71                entry_path_mapper: HashMap::new(),
72            });
73        }
74
75        Err(JsError::new("invalid arguments"))
76    }
77
78    #[wasm_bindgen(getter)]
79    pub fn length(&self) -> usize {
80        self.state.borrow().pak.entries.len()
81    }
82
83    #[wasm_bindgen]
84    pub fn entry(&mut self, index: usize) -> JsValue {
85        match self.state.borrow().pak.entries.get(index) {
86            Some(entry) => {
87                let entry_path = entry.path();
88                let state = match self.entry_path_mapper.entry(entry_path.to_vec()) {
89                    HashMapEntry::Vacant(entry) => {
90                        // Create a new state
91                        let state = Rc::new(EntryState {
92                            path: entry_path.to_vec(),
93                            deleted: false,
94                        });
95
96                        entry.insert(Rc::downgrade(&state));
97
98                        state
99                    }
100                    HashMapEntry::Occupied(mut entry) => {
101                        let weak_state = entry.get();
102                        match weak_state.upgrade() {
103                            Some(state) => state,
104                            None => {
105                                let state = Rc::new(EntryState {
106                                    path: entry_path.to_vec(),
107                                    deleted: false,
108                                });
109
110                                entry.insert(Rc::downgrade(&state));
111
112                                state
113                            }
114                        }
115                    }
116                };
117
118                let entry = Entry {
119                    pak_state: self.state.clone(),
120                    state,
121                };
122                entry.into()
123            }
124            None => JsValue::UNDEFINED,
125        }
126    }
127}

129struct EntryState {

131    path: Vec<u8>,

133    /// Whether this was deleted
134    deleted: bool,

137#[wasm_bindgen]
138pub struct Entry {
139    pak_state: Rc<RefCell<PakState>>,
140
141    state: Rc<EntryState>,
142}
143
144impl Entry {
145    fn path_raw_ref(&self) -> &[u8] {
146        &self.state.path
147    }
148
149    fn access_entry<F, T>(&self, func: F) -> Result<T, JsError>
150    where
151        F: FnOnce(&popcap_pak::Entry) -> T,
152    {
153        if self.state.deleted {
154            return Err(JsError::new("entry deleted from PAK file"));
155        }
156
157        let mut pak_state = self.pak_state.borrow_mut();
158        let entry_idx = pak_state
159            .get_entry_idx_by_path(self.path_raw())
160            .ok_or_else(|| JsError::new("failed to locate entry in PAK file"))?;
161        let entry = pak_state
162            .pak
163            .entries
164            .get(entry_idx)
165            .ok_or_else(|| JsError::new("internal bug"))?;
166        Ok(func(entry))
167    }
168}

170#[wasm_bindgen]
171impl Entry {
172    pub fn path_raw(&self) -> Vec<u8> {
173        self.path_raw_ref().into()
174    }
175
176    pub fn path(&self) -> Result<String, JsError> {
177        std::str::from_utf8(self.path_raw_ref())
178            .map(|s| s.into())
179            .map_err(|e| JsError::new(&e.to_string()))
180    }
181
182    pub fn path_lossy(&self) -> String {
183        String::from_utf8_lossy(self.path_raw_ref()).into()
184    }
185
186    pub fn name_raw(&self) -> Result<Vec<u8>, JsError> {
187        self.access_entry(|entry| entry.name().to_vec())
188    }
189
190    pub fn name(&self) -> Result<String, JsError> {
191        String::from_utf8(self.name_raw()?).map_err(|e| JsError::new(&e.to_string()))
192    }
193
194    pub fn name_lossy(&self) -> Result<String, JsError> {
195        Ok(String::from_utf8_lossy(&self.name_raw()?).into())
196    }
197
198    #[wasm_bindgen(js_name = "readFileData")]
199    pub fn read_file_data(&self) -> Result<Vec<u8>, JsError> {
200        self.access_entry(|entry| entry.iter_data().collect())
201    }
